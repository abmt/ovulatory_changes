library(brms)
load("full_data.rdata")
library(lubridate)
library(dplyr)

diary_long = diary %>% group_by(person) %>%
  filter(!is.na(included_all), !is.na(fertile_fab)) %>%
  filter(included_all != "cycling") %>%
  mutate(included = included_all,
         fertile_mean = mean(fertile_fab, na.rm = T)) %>%
  select(person, menstruation, fertile_mean, fertile_fab, included, extra_pair_2, extra_pair_3, extra_pair_4, extra_pair_5, extra_pair_6, extra_pair_7, extra_pair_8, extra_pair_9, extra_pair_10, extra_pair_11, extra_pair_12, extra_pair_13) %>%
  tidyr::gather(variable, value, -person, -menstruation, -fertile_mean, -fertile_fab, -included)

model = brm(value ~ menstruation + fertile_fab + fertile_mean + (1 + menstruation + fertile_fab|person) + (1|variable),
            data = diary_long,
            prior = c(set_prior("student_t(3, 0, 1)", class = "b"),
                      set_prior("cauchy(0, 2)", class = "sd")),
            chains = 10,
            cores = 10,
            iter = 2000,
            family = cumulative(link = "logit"),
            control = list(adapt_delta = 0.9)
)

saveRDS(model, file = "3_brms_extra_pair_long_ranef_hc.rds")

summary(model)
